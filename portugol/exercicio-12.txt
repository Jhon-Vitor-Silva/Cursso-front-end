programa {
  /*
EXERCÍCIO 12- O custo ao consumidor de um carro novo é a soma do custo de 
fábrica com a percentagem do distribuidor e dos impostos (aplicados, primeiro os impostos sobre o custo
 de fábrica, e depois a percentagem do distribuidor sobre o resultado). Supondo que a percentagem 
 do distribuidor seja de 28% e os impostos 45%. Escrever um algoritmo que leia o custo de fábrica de um carro 
 e informe o custo ao consumidor do mesmo.

  */
  funcao inicio() {
    real precofabrica, precoimposto, precodistribuidor

    escreva("Me fale o custo de fábrica: ")
    leia(precofabrica)
    precoimposto = precofabrica + (precofabrica * 0.45)

    precodistribuidor = precoimposto + (precoimposto * 0.28)

    escreva("O preço do consumidor sera: ", precodistribuidor)


    
  }
}
